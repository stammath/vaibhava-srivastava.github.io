ConvEpoc2HM <- function(x) {
  # for a given epoc number, get its hour:min
  yy <- x*15
  y.hr <- yy %/% 60
  y.min <- yy %% 60
  x <- paste(str_pad(y.hr, 2, side = 'left', pad='0'), 
                     str_pad(y.min, 2, side = 'left', pad='0'), 
                     '00', sep = ':')
}


df_RMA$Hour1 <- ConvEpoc2HM(df_RMA$EPOCH15)
DATE4 <- paste(strptime(df_RMA$date, format = "%m%d%Y", tz =""), df_RMA$Hour1, sep = ' ')
df_RMA$PCT_TIME <- as.POSIXct(DATE4, tz ="", format = "%Y-%m-%d %H:%M:%OS")
df_RMA$Hour <- strftime(df_RMA$PCT_TIME, format="%H")
df_RMA$DOW <- wday(df_RMA$PCT_TIME, label = TRUE)
